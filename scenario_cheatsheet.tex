\section{Scenario Quick Cards}
\zhint{每个场景的要点速查}

% --- BUSINESS / PREDICTION ---
\subsection{A. Prediction \& Forecasting}

\scenario{A1. Lottery / Mark Six Forecast}{彩票预测}
{Predict next draw using historical data, probability, regression.}
{Random by nature; no guaranteed pattern; overfitting to noise; small sample relative to combination space ($C_{49}^{6}$=13.98M).}
{Historical draw results from official website. Tabular: draw\_date, 6 winning numbers. Time series format.}

\scenario{A2. Stock Price Prediction}{股票预测}
{Predict future price/trend using LSTM on historical OHLCV data.}
{Black swans, market psychology, reflexivity, overfitting, non-stationary data.}
{yfinance API. Time series: Open, High, Low, Close, Volume. Sliding window (e.g. 30 days) as input.}

\scenario{A3. Automated Trading Agent}{自动交易}
{DQN agent learns buy/sell/hold strategy to maximize portfolio return.}
{Sparse reward, large state space, market regime changes, transaction costs.}
{yfinance API. State: prices + portfolio. Actions: buy/sell/hold. Reward: profit/loss.}

% --- RECOMMENDER ---
\subsection{B. Recommender Systems}

\scenario{B1. Movie/Music Recommendation}{电影/音乐推荐}
{Recommend items based on user preferences (Netflix/Spotify).}
{Cold start (new users/items), data sparsity, scalability, over-specialization.}
{User-item rating matrix (sparse). User profiles, item metadata (genre, tags). Interaction logs (clicks, plays, ratings).}

\scenario{B2. E-commerce Product Recommendation}{电商推荐}
{Recommend products to increase conversion rate (Amazon).}
{Same as B1 + fast-changing inventory, session-based behavior.}
{Purchase history, browsing logs, product features. Tabular + text (reviews).}

% --- NLP ---
\subsection{C. NLP Scenarios}

\scenario{C1. Fake News Detection}{假新闻检测}
{Classify news articles as real or fake using NLP models.}
{Polysemy, context-dependent meaning, evolving language, adversarial manipulation.}
{News articles with labels. Text data. Tokenized $\to$ embeddings. Classic: TF-IDF (25k$\times$5k). DL: BERT tokenizer (max\_length=256).}

\scenario{C2. Sentiment Analysis (Movie Reviews)}{情感分析}
{Classify reviews as positive/negative (IMDB dataset).}
{Sarcasm, negation, context dependency, word order matters.}
{IMDB: 25k train + 25k test. Text $\to$ tokenized $\to$ embeddings. Labels: pos/neg.}

\scenario{C3. Chatbot / Assistant}{聊天机器人}
{Conversational AI for customer service using GPT/LLaMA.}
{Intent understanding, multi-turn context, hallucination.}
{Conversation logs, FAQ pairs (query, response). Fine-tune GPT/LLaMA.}

\scenario{C4. Translation / Summarization}{翻译/摘要}
{Translate or summarize text using seq2seq Transformer (T5).}
{Word order, idioms, coherence, hallucination, rare languages.}
{Parallel corpus or document corpus. Text $\to$ tokenized. Markov chain as non-DL baseline.}

% --- AUDIO ---
\subsection{D. Audio / Speech}

\scenario{D1. Vocal Separation (Karaoke)}{人声分离}
{Isolate vocals from music for karaoke or remixing.}
{Frequency overlap between voice and instruments, harmonic complexity, temporal masking.}
{Audio files (WAV/MP3). Waveform (1D) or spectrogram (2D). Use librosa or Demucs.}

\scenario{D2. Text-to-Speech (TTS)}{语音合成}
{Convert text to natural-sounding speech (online dictionary, movie effects).}
{Natural prosody, emotion expression, speaker identity, robotic sound.}
{Text + paired audio recordings. Text $\to$ mel-spectrogram $\to$ waveform. Tacotron2 + WaveNet. Evaluate with MOS.}

\scenario{D3. Speech Recognition (STT)}{语音识别}
{Convert speech to text (smart home, voice assistant).}
{Accents, background noise, speaker variation, homophones.}
{Audio recordings + transcriptions. Waveform $\to$ features. Use HuBERT / Wav2Vec 2.0. Speaker-dependent vs independent.}

% --- VISION ---
\subsection{E. Computer Vision}

\scenario{E1. Accident / Anomaly Detection (CCTV)}{事故/异常检测}
{Detect falls, choking, violence in elderly homes / public areas via CCTV.}
{Real-time requirement, occlusion, lighting changes, false positives, privacy.}
{CCTV video frames. Image sequences (H$\times$W$\times$3). CNN for detection (ResNet), action recognition. Labels: normal/accident.}

\scenario{E2. Medical Image Diagnosis}{医疗影像诊断}
{Detect diseases (fractures, tumors) from X-ray/CT images.}
{Small dataset, class imbalance (rare diseases), need expert labels, interpretability required.}
{Hospital image database. Images (DICOM format $\to$ pixel arrays). Labels from radiologists. Use pre-trained CNN + fine-tune.}

\scenario{E3. Traffic Surveillance}{交通监控}
{Detect vehicles, accidents, violations from cameras.}
{Real-time, weather/lighting, occlusion, diverse vehicle types.}
{Camera feeds. Video frames. Object detection (Mask R-CNN). Labels: vehicle/violation type.}

\scenario{E4. Image Captioning}{图像描述}
{Generate text for images (accessibility for visually impaired).}
{Identify objects+actions+relationships+grammar. Feature bottleneck.}
{Image-caption pairs (COCO). 224$\times$224$\times$3 + text. CNN+RNN or ViT+Transformer. Eval: BLEU, CIDEr.}

% --- IMAGE/VIDEO GENERATION ---
\subsection{F. Generation}

\scenario{F1. AI Art / Marketing Image Generation}{AI艺术/营销图生成}
{Generate images from text descriptions for design/advertising.}
{High dimensionality (pixels), must imagine background, copyright concerns, quality control.}
{Text-image pairs. Text prompt $\to$ DALL-E/Stable Diffusion. Evaluate with FID. CLIP ranks outputs.}

\scenario{F2. DeepFake Detection}{深度伪造检测}
{Detect manipulated videos to prevent fraud/misinformation.}
{Increasingly realistic fakes, best AI only 80\% detection, adversarial attacks.}
{Real + fake video pairs. Video frames (face crops). Binary classification. Features: blinking, lip sync, skin texture.}

\scenario{F3. 2D-to-3D Reconstruction}{2D转3D}
{Generate 3D model from a single 2D image (game assets, architecture).}
{Infinite 3D shapes can explain one 2D view; needs strong priors; floater artifacts.}
{RGB images + 3D ground truth (Objaverse). Image $\to$ TripoSR (DINOv1 + Transformer + NeRF $\to$ mesh). Eval: Chamfer Distance, F-Score.}

\scenario{F4. Text-to-3D Generation}{文字转3D}
{Generate 3D models from text descriptions (Shap-E).}
{Lower fidelity than manual modeling, topological errors, floaters, bottlenecked by 2D model.}
{Text-3D pairs. Text prompt $\to$ image diffusion $\to$ point cloud diffusion $\to$ mesh. $\sim$13s per asset.}

% --- GAME / RL ---
\subsection{G. Game \& RL}

\scenario{G1. Card Game AI (Blackjack)}{21点AI}
{Train agent to play Blackjack optimally using RL.}
{Sparse reward (only at end), stochastic environment, exploration vs exploitation.}
{Simulated games. State: (player\_sum, dealer\_card, usable\_ace). Actions: hit/stand. Use D3QN, A3C, or AlphaZero+MCTS.}

\scenario{G2. Board Game AI (Chess/Go)}{棋类AI}
{Master board games via self-play (AlphaZero).}
{Huge state space, long-term planning, MCTS cost.}
{Self-play games. State: board position. MCTS + dual-headed net (policy+value).}

\scenario{G3. Multi-Game Agent}{多游戏智能体}
{Single agent plays multiple games without forgetting.}
{Catastrophic forgetting: new task overwrites old weights.}
{Multiple game envs. EWC protects important weights. Train A $\to$ B(+EWC) $\to$ C.}

\subsection{H. Robot Learning}

\scenario{H1. Robot Locomotion}{机器人行走}
{Train humanoid to walk using RL + transfer learning.}
{High-dim continuous actions, balance, sim-to-real gap.}
{Simulated env. State: joint angles. Transfer: crawl $\to$ wall-walk $\to$ free-walk.}

\scenario{H2. Robot Manipulation}{机器人抓取}
{Train robot arm to grasp objects.}
{Object variety, precision, deformable objects.}
{Camera + joint data. Imitation / inverse RL / diffusion policies.}
