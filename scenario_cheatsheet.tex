% --- BUSINESS / PREDICTION ---
\subsection{A. Prediction \& Forecasting}

\scenario{A1. Lottery / Mark Six}{彩票预测}
{Use past draw data to predict future numbers with probability and regression models.}
{Totally random; no real pattern; easy to overfit noise; huge number of combinations ($C_{49}^{6}$=13.98M).}
{Past draw results from official site. Table: date + 6 numbers. Time series format.}

\scenario{A2. Stock Price Prediction}{股票预测}
{Use LSTM on past stock data (OHLCV) to predict future price or trend.}
{Sudden crashes (black swan), human emotions affect market, data keeps changing, easy to overfit.}
{yfinance API. Time series: Open, High, Low, Close, Volume. Use 30-day sliding window as input.}

\scenario{A3. Auto Trading Agent}{自动交易}
{Train a DQN agent to learn when to buy, sell, or hold to make profit.}
{Reward is sparse (only see profit/loss later), market changes over time, trading fees.}
{yfinance API. State = prices + what we own. Action = buy/sell/hold. Reward = profit or loss.}

% --- RECOMMENDER ---
\subsection{B. Recommender Systems}

\scenario{B1. Movie/Music Recommendation}{电影/音乐推荐}
{Suggest movies or songs that a user will like (Netflix/Spotify style).}
{Cold start: new users have no history; data is very sparse; users may only like one type.}
{User-item rating table (mostly empty). Also: user profiles, item info (genre, tags), click/play logs.}

\scenario{B2. E-commerce Recommendation}{电商推荐}
{Suggest products to help users find what they want (Amazon style).}
{Same as B1, plus products change fast, users browse quickly.}
{Purchase history, browsing logs, product details. Mix of table data + text (reviews).}

% --- NLP ---
\subsection{C. NLP Scenarios}

\scenario{C1. Fake News Detection}{假新闻检测}
{Tell if a news article is real or fake using NLP.}
{Same word can mean different things; language keeps changing; people try to trick the system.}
{News articles with real/fake labels. Text data. Classic: TF-IDF (25k$\times$5k). DL: BERT tokenizer (max 256 tokens).}

\scenario{C2. Sentiment Analysis}{情感分析}
{Tell if a movie review is positive or negative (IMDB).}
{Sarcasm is hard to detect; ``not good'' vs ``good''; word order matters.}
{IMDB: 25k train + 25k test. Text $\to$ tokens $\to$ embeddings. Two labels: pos/neg.}

\scenario{C3. Chatbot / Assistant}{聊天机器人}
{Build an AI that can talk to customers and answer questions.}
{Must understand what user wants; remember past messages; may give wrong answers (hallucination).}
{Chat logs, FAQ pairs (question + answer). Fine-tune GPT or LLaMA.}

\scenario{C4. Translation / Summary}{翻译/摘要}
{Translate text between languages, or make a short summary.}
{Different languages have different word orders; idioms hard to translate; output must make sense.}
{Paired sentences (source + target) or documents. Text $\to$ tokens. Markov chain as simple baseline.}

% --- AUDIO ---
\subsection{D. Audio / Speech}

\scenario{D1. Vocal Separation (Karaoke)}{人声分离}
{Split a song into voice and music for karaoke.}
{Voice and instruments share same frequency range; hard to separate cleanly.}
{Audio files (WAV/MP3). Sound wave (1D) or spectrogram (2D). Use librosa or Demucs model.}

\scenario{D2. Text-to-Speech (TTS)}{语音合成}
{Make computer read text out loud like a real person.}
{Must sound natural, not robotic; show emotion; keep speaker identity.}
{Text + matching audio recordings. Pipeline: text $\to$ mel-spectrogram $\to$ sound wave. Use Tacotron2 + WaveNet. Eval: MOS score.}

\scenario{D3. Speech-to-Text (STT)}{语音识别}
{Turn spoken words into text (for smart home, voice assistant).}
{Different accents, noisy rooms, people talk differently, words that sound the same.}
{Audio recordings + written transcripts. Sound wave $\to$ features. Use HuBERT or Wav2Vec 2.0.}

% --- VISION ---
\subsection{E. Computer Vision}

\scenario{E1. Accident Detection (CCTV)}{事故检测}
{Watch CCTV to detect falls, choking in elderly homes or public places.}
{Must work in real-time; things can block the view; lighting changes; false alarms; privacy issues.}
{CCTV video frames. Images (H$\times$W$\times$3). Use CNN (ResNet) to classify. Labels: normal or accident.}

\scenario{E2. Medical Image Diagnosis}{医疗影像}
{Find diseases (broken bones, tumors) in X-ray or CT images.}
{Very few sick cases (imbalanced); need doctor labels; must be explainable; small dataset.}
{Hospital image database. Medical images $\to$ pixel arrays. Doctor labels. Use pre-trained CNN + fine-tune.}

\scenario{E3. Traffic Surveillance}{交通监控}
{Detect cars, accidents, and violations from traffic cameras.}
{Must be real-time; bad weather/light; cars block each other; many vehicle types.}
{Traffic camera video. Video frames. Use object detection (Mask R-CNN). Labels: car type, violation.}

\scenario{E4. Image Captioning}{图像描述}
{Write a sentence to describe what is in a picture (help blind people).}
{Must find objects + actions + how they relate + write correct grammar.}
{Image + caption pairs (COCO dataset). Image: 224$\times$224$\times$3. Use CNN+RNN or ViT+Transformer. Eval: BLEU, CIDEr.}

% --- IMAGE/VIDEO GENERATION ---
\subsection{F. Generation}

\scenario{F1. AI Image Generation}{AI图片生成}
{Create pictures from text descriptions for art or ads.}
{Output is very high-dim (millions of pixels); must fill in background; copyright issues.}
{Text-image pairs. Type text prompt $\to$ use DALL-E or Stable Diffusion. Eval: FID. CLIP scores the results.}

\scenario{F2. DeepFake Detection}{深度伪造检测}
{Find fake videos to stop scams and misinformation.}
{Fakes look more and more real; best AI only catches 80\%; attackers keep improving.}
{Real + fake video pairs. Face crops from video frames. Binary: real or fake. Look at: blinking, lip sync, skin.}

\scenario{F3. 2D to 3D Model}{2D转3D}
{Turn a single photo into a 3D model (for games, architecture).}
{Many possible 3D shapes match one 2D view; needs learned priors; may have floating artifacts.}
{RGB images + 3D ground truth. Use TripoSR: image $\to$ encoder $\to$ transformer $\to$ NeRF $\to$ mesh. Eval: Chamfer Distance, F-Score.}

\scenario{F4. Text to 3D Model}{文字转3D}
{Type words and get a 3D model (Shap-E by OpenAI).}
{Less detailed than hand-made models; shape errors; floating points; limited by 2D image quality.}
{Text + 3D pairs. Prompt $\to$ image diffusion $\to$ point cloud diffusion $\to$ mesh. About 13 seconds each.}

% --- GAME / RL ---
\subsection{G. Game \& RL}

\scenario{G1. Card Game AI (Blackjack)}{21点AI}
{Train an AI to play Blackjack well using RL.}
{Only get reward at game end (sparse); cards are random; must balance try-new vs use-best.}
{Simulated games. State = (my total, dealer card, usable ace). Actions: hit or stand. Use D3QN, A3C, or AlphaZero.}

\scenario{G2. Board Game AI}{棋类AI}
{Train AI to master Chess or Go by playing against itself (AlphaZero).}
{Huge number of possible positions; need to plan many moves ahead; MCTS is expensive.}
{Self-play generated games. State = board position. Use MCTS + dual net (policy head + value head).}

\scenario{G3. Multi-Game Agent}{多游戏智能体}
{One AI that can play many different games without forgetting old ones.}
{Catastrophic forgetting: learning a new game erases what it learned before.}
{Multiple game environments. Use EWC to protect important weights. Train: game A $\to$ game B (with EWC) $\to$ game C.}

\subsection{H. Robot Learning}

\scenario{H1. Robot Walking}{机器人行走}
{Teach a robot to walk like a human using RL + transfer learning.}
{Many joints to control at once; must keep balance; simulation differs from real world.}
{Simulated physics environment. State = joint angles + body pose. Learn step by step: crawl $\to$ walk with wall $\to$ walk freely.}

\scenario{H2. Robot Grasping}{机器人抓取}
{Teach a robot arm to pick up objects.}
{Objects come in all shapes; needs high precision; soft objects are tricky.}
{Camera images + robot arm joint data. Use imitation learning, inverse RL, or diffusion policies.}
