\section{Key Terminology}
% ══════════════════════════════════════════════════════════

\subsection{Data-Related Terms}

\begin{defbox}[Definitions]
\textbf{Database:} A collection of data that is well-organized and easily managed.

\textbf{Data Analysis:} A process of retrieving knowledge from data.

\textbf{Data Mining:} A process of extracting knowledge in \underline{large data sets}.

\textbf{Big Data:} A field that handles \underline{large and complex} data sets.\\
Challenges: storage, analysis, transfer, search, visualization, updating.
\end{defbox}

\subsection{AI / ML / DL}

\begin{infobox}[AI $\supset$ ML $\supset$ DL]
\textbf{AI} (Artificial Intelligence):\\
Knowledge or behaviour demonstrated by machines.

\textbf{ML} (Machine Learning):\\
Computer algorithms that improve knowledge retrieval from data. \textit{A part of AI.}

\textbf{DL} (Deep Learning):\\
Use of \underline{multiple layers} in the network to improve knowledge retrieval from data. \textit{A part of ML.}
\end{infobox}

\subsection{Three Data Roles}

\begin{defbox}
\textbf{Data Scientist}:\\
Transforms data to valuable \underline{insights \& strategic decisions}.

\textbf{ML Engineer}:\\
Turns data-driven models into \underline{practical applications}.

\textbf{Data Engineer}:\\
Works on the foundation of a \underline{robust data infrastructure}.
\end{defbox}

\subsection{What is Data Science?}

\begin{infobox}
Uses \textbf{scientific methods}, processes, algorithms, and systems to \textbf{extract knowledge and insights} from data across a broad range of application domains.

\smallskip
\textbf{Relevant disciplines:}
\begin{itemize}
  \item Computer Science -- database, processing, programming
  \item Informatics -- computation systems, knowledge extraction
  \item Statistics -- interpretation of results
  \item Other disciplines -- depend on problem definitions
  \item Communication and Visualization
\end{itemize}
\end{infobox}

\section{Types of Data Analytics}
% ══════════════════════════════════════════════════════════

\begin{defbox}[4 Types]
\textbf{Predictive}: ``What \underline{might happen} in the future?''

\textbf{Descriptive}: ``What \underline{happened}?''

\textbf{Prescriptive}: ``What \underline{should we do} next?''

\textbf{Diagnostic}: ``\underline{Why} did this happen?''
\end{defbox}

\begin{exbox}[Example: Ice-cream Shop]
\textbf{Predictive}: How much chocolate ice-cream will be needed \underline{tomorrow}?

\textbf{Descriptive}: What are the top flavours in the \underline{past week}?

\textbf{Prescriptive}: \underline{Should} we increase or reduce quality to make more money?

\textbf{Diagnostic}: \underline{Why} are chocolate and vanilla the best sellers?
\end{exbox}

\section{Application Examples}
% ══════════════════════════════════════════════════════════

\begin{defbox}
\textbf{1. Handwritten Digit Recognition}\\
MNIST dataset; one of the first widely used ML systems.

\textbf{2. Recommender Systems}\\
Netflix Prize competition; winner: BellKor's Pragmatic Chaos. Recommend movies/books based on past behaviour.

\textbf{3. Face Detection / Recognition}\\
Detect and recognize human faces. E.g. Microsoft Face API.

\textbf{4. Machine Translation}\\
SMT (Statistical) or NMT (Neural) machine translation. E.g. Google NMT -- analyse parallel text corpora.

\textbf{5. Face Generator}\\
GAN (Generative Adversarial Network) generates fake but realistic images by learning features from existing images.

\textbf{6. ChatGPT}\\
NLP tool driven by AI; human-like conversations; compose emails, essays, and code.
\end{defbox}

\section{EU AI Act}
% ══════════════════════════════════════════════════════════

\begin{infobox}[First Regulation on AI]
\textbf{Goals:}
\begin{itemize}
  \item Ensure AI systems are \textbf{safe} and respect \textbf{fundamental rights}
  \item Foster investment and innovation
  \item Enhance governance and enforcement
  \item Encourage a single EU market for AI
\end{itemize}
\end{infobox}

\subsection{4 Risk Levels}

\begin{warnbox}[Unacceptable Risk -- \textit{Prohibited}]
\begin{itemize}
  \item Voice-activated toys encouraging dangerous behaviour in children
  \item Biometric identification
  \item Social scoring
  \item Deceptive techniques
  \item Facial recognition databases
  \item \textit{Exception: law enforcement}
\end{itemize}
\textbf{Strategy:} Full prohibition; legal liability for development/deployment.
\end{warnbox}

\begin{warnbox}[High Risk -- \textit{Conformity Assessment}]
\begin{itemize}
  \item AI in cars, medical devices
  \item AI in education, vocational training, employment
  \item AI for record-keeping
\end{itemize}
\textbf{Strategy:} Conformity assessments; human-in-the-loop requirements; real-time monitoring.
\end{warnbox}

\begin{infobox}[Limited Risk -- \textit{Transparency}]
\begin{itemize}
  \item ChatGPT, Deep Fakes, Chat bots
\end{itemize}
\textbf{Strategy:} Transparency obligations; user opt-out mechanisms; annual compliance audits.
\end{infobox}

\begin{infobox}[Minimal Risk -- \textit{Code of Conduct}]
\begin{itemize}
  \item AI video games, spam filters
\end{itemize}
\textbf{Strategy:} Self-certification.
\end{infobox}

\subsection{12 Ethical AI Principles}

\begin{defbox}
\begin{enumerate}[leftmargin=1.2em]
  \item \textbf{Transparency \& Interpretability}:\\
    Decision processes explainable to humans clearly.
  \item \textbf{Reliability, Robustness \& Security}:\\
    Consistent results; handle errors; secure against cyber-attacks.
  \item \textbf{Fairness}:\\
    No favouritism or discrimination; respect individuals behind data.
  \item \textbf{Diversity \& Inclusion}:\\
    Respect interests of all stakeholders.
  \item \textbf{Human Oversight}:\\
    Intervention level dictated by severity of ethical issues.
  \item \textbf{Lawfulness \& Compliance}:\\
    Act in accordance with law and regulations.
  \item \textbf{Safety}:\\
    Must not compromise physical safety or mental integrity.
  \item \textbf{Data Privacy}:\\
    Right to be informed of data collection purpose; data not kept longer than necessary; protect personal data.
  \item \textbf{Accountability}:\\
    Clearly identifiable accountable party.
  \item \textbf{Beneficial AI}:\\
    Promote the common good.
  \item \textbf{Cooperation \& Openness}:\\
    Multi-stakeholder open cooperation in AI ecosystem.
  \item \textbf{Sustainability \& Just Transition}:\\
    Mitigation for societal and environmental impacts.
\end{enumerate}
\end{defbox}

\subsection{AI Governance Model}

\begin{infobox}[Three Lines of Defence]
A layered governance structure ensuring accountability at multiple levels within an organisation for AI systems.
\end{infobox}

\subsection{AI Lifecycle \& Impact Assessment}

\begin{exbox}
\textbf{AI Lifecycle} aligned with \textbf{SDLC}:\\
Project Request $\to$ Analysis \& Design $\to$ Implementation $\to$ Maintenance

\smallskip
\textbf{Impact Assessment Stages:}
\begin{enumerate}
  \item \textbf{Project Request}: Categorise as high risk or not via ``Risk Gating Questions''.
  \item \textbf{Design \& Implementation}: Conduct AI application impact assessment.
  \item \textbf{Maintenance}: Review, update assessment; escalate significant issues.
\end{enumerate}
\end{exbox}
