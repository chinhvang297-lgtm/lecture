\section{1. Key Terminology}
% ══════════════════════════════════════════════════════════

\subsection{1.1 Data-Related Terms}

\begin{defbox}[Definitions]
\begin{itemize}
  \item \textbf{Database}: A collection of data that is well-organized and easily managed.
  \item \textbf{Data Analysis}: A process of retrieving knowledge from data.
  \item \textbf{Data Mining}: A process of extracting knowledge in \underline{large} data sets.
  \item \textbf{Big Data}: A field that handles large and complex data sets.
    \begin{itemize}[leftmargin=1em]
      \item Challenges: storage, analysis, transfer, search, visualization, updating.
    \end{itemize}
\end{itemize}
\end{defbox}

\subsection{1.2 AI / ML / DL}

\begin{infobox}[Relationship: AI $\supset$ ML $\supset$ DL]
\begin{itemize}
  \item \textbf{Artificial Intelligence (AI)}: Knowledge or behaviour demonstrated by machines.
  \item \textbf{Machine Learning (ML)}: Computer algorithms that improve knowledge retrieval from data. \textit{A part of AI.}
  \item \textbf{Deep Learning (DL)}: Use of multiple layers in the network to improve knowledge retrieval. \textit{A part of ML.}
\end{itemize}
\end{infobox}

\subsection{1.3 Data Roles}

\begin{defbox}[Three Data Roles]
\begin{itemize}
  \item \textbf{Data Scientist}: Transforms data to valuable \underline{insights and strategic decisions}.
  \item \textbf{ML Engineer}: Turns data-driven models into \underline{practical applications}.
  \item \textbf{Data Engineer}: Works on the foundation of a \underline{robust data infrastructure}.
\end{itemize}
\end{defbox}

\subsection{1.4 Data Science}

\begin{infobox}[What is Data Science?]
Uses \textbf{scientific methods}, processes, algorithms, and systems to \textbf{extract knowledge and insights} from data across a broad range of application domains.

\smallskip
\textbf{Relevant disciplines:}
\begin{itemize}
  \item Computer Science -- database, data processing, programming
  \item Informatics -- computation systems, knowledge extraction
  \item Statistics -- interpretation of results
  \item Other disciplines -- depend on problem definitions
  \item Communication and Visualization
\end{itemize}
\end{infobox}

\section{2. Data Science Lifecycle}
% ══════════════════════════════════════════════════════════

\begin{warnbox}[5 Stages (Must Know!)]
\begin{enumerate}
  \item \textbf{Data Acquisition} -- collect/obtain data
  \item \textbf{Data Cleaning \& Warehousing} -- fix errors, store properly

\section{3. Types of Data Analytics}
% ══════════════════════════════════════════════════════════

\begin{defbox}[4 Types of Analytics]
\renewcommand{\arraystretch}{1.3}
\begin{tabularx}{\linewidth}{@{}l X@{}}
\toprule
\textbf{Type} & \textbf{Core Question} \\
\midrule
\textbf{Predictive} & ``What \underline{might happen} in the future?'' \\
\textbf{Descriptive} & ``What \underline{happened} / is happening?'' \\
\textbf{Prescriptive} & ``What \underline{should we do} next?'' \\
\textbf{Diagnostic} & ``\underline{Why} did this happen?'' \\
\bottomrule
\end{tabularx}
\end{defbox}

\begin{exbox}[Example: Ice-cream Shop]
\begin{itemize}
  \item \textbf{Predictive}: How much chocolate ice-cream will be needed \underline{tomorrow}?
  \item \textbf{Descriptive}: What are the top flavours in the \underline{past week}?
  \item \textbf{Prescriptive}: \underline{Should} we increase or reduce quality to make more money?
  \item \textbf{Diagnostic}: \underline{Why} are chocolate and vanilla the best sellers?
\end{itemize}
\end{exbox}

\subsection{6.1 Risk Levels (EU AI Act)}

\begin{defbox}[4 Risk Tiers]
\renewcommand{\arraystretch}{1.25}
\begin{tabularx}{\linewidth}{@{}>{\bfseries}l X@{}}
\toprule
\textbf{Level} & \textbf{Details} \\
\midrule
Unacceptable (Prohibited) &
  \begin{itemize}[nosep,leftmargin=0.8em]
    \item Voice toys encouraging dangerous behaviour in children
    \item Biometric identification
    \item Social scoring
    \item Deceptive techniques

    \item Exception: law enforcement
  \end{itemize} \\

High (Conformity Assessment) &
  \begin{itemize}[nosep,leftmargin=0.8em]

  \end{itemize} \\

Limited (Transparency) &
  \begin{itemize}[nosep,leftmargin=0.8em]
    \item ChatGPT, Deep Fakes, Chat bots
    \item Must inform users they interact with AI
  \end{itemize} \\
\midrule
Minimal (Code of Conduct) &
  \begin{itemize}[nosep,leftmargin=0.8em]
    \item AI video games, spam filters
  \end{itemize} \\
\bottomrule
\end{tabularx}

\section{7. HK Ethical AI Framework}
% ══════════════════════════════════════════════════════════

\begin{infobox}[Background]
Issued by the \textbf{Digital Policy Office (DPO)} in \textbf{July 2024}. Internal adoption within HK Government for AI and big data analytics applications.
\end{infobox}

\subsection{7.1 Twelve Ethical AI Principles}

\begin{defbox}[12 Principles]
\begin{enumerate}[leftmargin=1.4em]
  \item \textbf{Transparency \& Interpretability}: Decision processes explainable to humans.
  \item \textbf{Reliability, Robustness \& Security}: Consistent results, handle errors, secure against cyber-attacks.
  \item \textbf{Fairness}: No favouritism or discrimination; respect individuals behind data.
  \item \textbf{Diversity \& Inclusion}: Respect interests of all stakeholders.
  \item \textbf{Human Oversight}: Degree of human intervention dictated by severity of ethical issues.
  \item \textbf{Lawfulness \& Compliance}: Act in accordance with law and regulations.
  \item \textbf{Safety}: Must not compromise physical safety or mental integrity.
  \item \textbf{Data Privacy}: Right to be informed of data collection purpose; data not kept longer than necessary; practicable steps to protect personal data.
  \item \textbf{Accountability}: Clearly identifiable accountable party (individual or organisation).
  \item \textbf{Beneficial AI}: Promote the common good.
  \item \textbf{Cooperation \& Openness}: Multi-stakeholder open cooperation.
  \item \textbf{Sustainability \& Just Transition}: Mitigation for societal and environmental impacts.
\end{enumerate}
\end{defbox}

\subsection{7.2 AI Governance -- Lines of Defence}

\begin{infobox}[Three Lines of Defence Model]
A layered governance structure ensuring accountability at multiple levels within an organisation for AI systems.
\end{infobox}

\subsection{7.3 AI Lifecycle}

\begin{exbox}[AI Lifecycle Overview]
AI lifecycle aligned with \textbf{System Development Lifecycle (SDLC)}:
\begin{itemize}
  \item Project Request $\to$ System Analysis \& Design $\to$ Implementation $\to$ Maintenance
\end{itemize}
\end{exbox}

\subsection{7.4 AI Impact Assessment Stages}

\begin{infobox}[Key Stages]
\begin{enumerate}
  \item \textbf{Project Request}: Categorise project as high risk or not via ``Risk Gating Questions''.
  \item \textbf{System Analysis/Design \& Implementation}: Conduct AI application impact assessment.
  \item \textbf{System Maintenance}: Review and update assessment; escalate significant issues.
\end{enumerate}
\end{infobox}
