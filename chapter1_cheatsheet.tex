\section{Key Terminology}
% ============================================================

\subsection{Data-related Terms}
\begin{itemize}[leftmargin=*, nosep, itemsep=1pt]
    \item \textbf{Database}: a collection of well-organized, easily managed data
    \item \textbf{Data Analysis}: retrieving knowledge from data
    \item \textbf{Data Mining}: extracting knowledge from \emph{large} data sets
    \item \textbf{Big Data}: handling large and complex data sets
    \begin{itemize}[nosep, itemsep=0pt]
        \item Challenges: storage, analysis, transfer, search, visualization, updating
    \end{itemize}
\end{itemize}

\subsection{AI / ML / DL}
\begin{itemize}[leftmargin=*, nosep, itemsep=1pt]
    \item \textbf{Artificial Intelligence (AI)}: knowledge or behaviour demonstrated by machines
    \item \textbf{Machine Learning (ML)}: algorithms that improve knowledge retrieval from data (subset of AI)
    \item \textbf{Deep Learning (DL)}: multiple layers in network for improved knowledge retrieval (subset of ML)
\end{itemize}

\keybox{
\textbf{Relationship}: AI $\supset$ ML $\supset$ DL
}

\subsection{Roles in Data Science}
\begin{itemize}[leftmargin=*, nosep, itemsep=1pt]
    \item \textbf{Data Scientist}: transforms data into valuable insights and strategic decisions
    \item \textbf{ML Engineer}: turns data-driven models into practical applications
    \item \textbf{Data Engineer}: builds foundation of robust data infrastructure
\end{itemize}

% ============================================================
\section{Data Science}
% ============================================================

Uses scientific methods, processes, algorithms, and systems to extract knowledge and insights from data.

\subsection{Relevant Disciplines}
\begin{itemize}[leftmargin=*, nosep, itemsep=0pt]
    \item Computer science --- database, data processing, programming
    \item Informatics --- computation systems, knowledge extraction
    \item Statistics --- interpretation of results
    \item Communication and visualization
    \item Domain-specific disciplines
\end{itemize}

\subsection{Data Science Lifecycle}
\begin{enumerate}[leftmargin=*, nosep, itemsep=1pt]
    \item \textbf{Data Acquisition} --- collect raw data
    \item \textbf{Data Cleaning \& Warehousing} --- preprocess and store
    \item \textbf{Data Modeling \& Summarization} --- build models
    \item \textbf{Data Analysis} --- extract insights
    \item \textbf{Data Visualization} --- present findings
\end{enumerate}

\subsection{Importance of Data Science}
\begin{enumerate}[leftmargin=*, nosep, itemsep=0pt]
    \item Cost efficient
    \item Decision making tool
    \item Understand customer behaviour
    \item Analyze and manage risks
    \item Product development and innovations
\end{enumerate}

\section{Classic Examples}
% ============================================================

\begin{enumerate}[leftmargin=*, nosep, itemsep=1pt]
    \item \textbf{Handwritten Digit Recognition} (MNIST) --- one of the first widely used ML systems

    \item \textbf{Face Generator} --- Generative Adversarial Networks (GANs)
    \item \textbf{ChatGPT} --- NLP tool for human-like conversations, composing emails, essays, and code
\end{enumerate}

\section{Types of Data Analytics}
% ============================================================

\begin{enumerate}[leftmargin=*, nosep, itemsep=1pt]
    \item \textbf{Predictive}: ``What might happen in the future?''
    \begin{itemize}[nosep, itemsep=0pt]
        \item E.g.\ How much chocolate ice-cream needed tomorrow?
    \end{itemize}
    \item \textbf{Descriptive}: ``What happened?''
    \begin{itemize}[nosep, itemsep=0pt]
        \item E.g.\ What are the top flavours last week?
    \end{itemize}
    \item \textbf{Prescriptive}: ``What should we do next?''
    \begin{itemize}[nosep, itemsep=0pt]
        \item E.g.\ Should we increase/reduce quality?
    \end{itemize}
    \item \textbf{Diagnostic} (optional): ``Why did this happen?''
    \begin{itemize}[nosep, itemsep=0pt]
        \item E.g.\ Why are chocolate \& vanilla best sellers?
    \end{itemize}
\end{enumerate}

\section{AI Ethics \& Governance}
% ============================================================

\subsection{EU AI Act}
First regulation on AI --- ensure AI systems in EU are safe and respect fundamental rights.

\begin{center}
\small
\begin{tabular}{lp{4.5cm}}
\toprule
\textbf{Risk Level} & \textbf{Examples \& Strategy} \\
\midrule
Unacceptable & Biometric ID, social scoring, deceptive techniques $\rightarrow$ \textbf{Prohibited} \\
High & Medical devices, autonomous vehicles, education AI $\rightarrow$ Conformity assessment \\
Limited & ChatGPT, deep fakes, chatbots $\rightarrow$ Transparency obligations \\
Minimal & AI video games, spam filters $\rightarrow$ Code of conduct \\
\bottomrule
\end{tabular}
\end{center}

\subsection{HK Ethical AI Framework (Jul 2024)}
Issued by Digital Policy Office (DPO) for internal government AI adoption.

\textbf{12 Ethical AI Principles}:
\begin{enumerate}[leftmargin=*, nosep, itemsep=0pt]
    \item Transparency \& Interpretability
    \item Reliability, Robustness \& Security
    \item Fairness (no discrimination)
    \item Diversity \& Inclusion
    \item Human Oversight
    \item Lawfulness \& Compliance
    \item Safety
    \item Data Privacy
    \item Accountability
    \item Beneficial AI
    \item Cooperation \& Openness
    \item Sustainability \& Just Transition
\end{enumerate}

\subsection{AI Governance --- Lines of Defence}
Multi-layered governance model aligning AI lifecycle with system development lifecycle, including risk gating and impact assessment at each stage.

\subsection{Mainland China}
``Global AI Governance Initiative'' and ``Interim Measures for Management of Generative AI Services'' (effective Aug 2023), under Cybersecurity Law, Data Security Law, and Personal Information Protection Law.
